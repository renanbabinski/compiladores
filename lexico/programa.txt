:STATEMENTS
argumento:INT
fatorial:INT

:ALGORITHM
% Calcula o fatorial de um nÃºmero inteiro
READ argumento
ASSIGN argumento TO fatorial
IF argumento = 0 THEN ASSIGN 1 TO fatorial
WHILE argumento > 1
    BEGIN
        ASSIGN fatorial * (argumento - 1) TO fatorial
        ASSIGN argumento - 1 TO argumento
    END
PRINT fatorial
